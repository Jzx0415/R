lack <- function(z){
  r<-0
  g<-0
  b<-0
  for(i in z){
    if(i<=50){
      r=1
    }
    if(i>50 || i<=70){
      g=1
    }
    if(i>70){
      b=1
    }
  }
  sum=r+g+b
  if(sum<3){
    return(TRUE)
  }
  else{
    return(FALSE)
  }
}
